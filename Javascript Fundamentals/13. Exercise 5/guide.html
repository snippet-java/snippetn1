Complete the below code snippet to print the grade on following criteria :
Range : Grade
<60 : F
<70 : D
<80 : C
<90 : B
<=100 : A 

 
var mark = 85;   
 
//Use if condition to compute the grade
if (mark < 60){  
    console.log("Grade : F");        
} else if (mark < 70) {  
    console.log("Grade : D");   
} else if (mark < 80) {  
    console.log("Grade : C");   
} else if (mark < 90) {  
    console.log("Grade : B");   
} else if (mark < 100) {  
    console.log("Grade : A");  
}
