//This function returns garde for a mark
function getGrade(mark) {
    if (mark < 60){  
      return "E";       
    }   
    else if (mark < 70) {  
      return "D";  
    }   
    else if (mark < 80) {  
      return "C";  
    } else if (mark < 90) {  
      return "B";   
    } else if (mark <= 100) {  
      return "A";  
    } else {
      return "Z";
    }
}
